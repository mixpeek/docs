## Prerequisites

Before you begin, make sure you have:

- A Mixpeek account (sign up at [mixpeek.com/start](https://mixpeek.com/start))
- Your API key (found in your dashboard)
- Basic familiarity with REST APIs

## Step 2: Create a Namespace

A namespace is an isolated environment for your data. Let's create one:

<CodeGroup>
  ```python Python
  # Create a namespace
  namespace = mp.namespaces.create(
      name="quickstart",
      description="My first Mixpeek project"
  )
  
  namespace_id = namespace["namespace_id"]
  print(f"Created namespace: {namespace_id}")
  ```

  ```javascript JavaScript
  // Create a namespace
  const namespace = await mp.namespaces.create({
    name: "quickstart",
    description: "My first Mixpeek project"
  });
  
  const namespaceId = namespace.namespace_id;
  console.log(`Created namespace: ${namespaceId}`);
  ```

  ```bash cURL
  curl -X POST "https://api.mixpeek.com/v1/namespaces" \
    -H "Authorization: Bearer $MIXPEEK_API_KEY" \
    -H "Content-Type: application/json" \
    -d '{
      "name": "quickstart",
      "description": "My first Mixpeek project"
    }'
  ```
</CodeGroup>

## Step 4: Create a Feature Extraction Pipeline

Let's define a pipeline that will extract features from our content:

<CodeGroup>
  ```python Python
  # Create a collection to store processed documents
  collection = mp.collections.create(
      namespace_id=namespace_id,
      name="content-collection",
      description="Processed content documents"
  )
  
  collection_id = collection["collection_id"]
  
  # Define a feature extraction pipeline
  pipeline = mp.pipelines.create(
      namespace_id=namespace_id,
      name="content-pipeline",
      description="Extract features from content",
      input_bucket_id=bucket_id,
      output_collection_ids=[collection_id],
      feature_extractors=[
          {
              "embed": [
                  {
                      "type": "url",
                      "embedding_model": "multimodal"
                  },
                  {
                      "type": "text",
                      "value": "describe this content",
                      "embedding_model": "text"
                  }
              ],
              "detect": {
                  "faces": {"enabled": True},
                  "objects": {"enabled": True}
              },
              "transcribe": {"enabled": True}
          }
      ]
  )
  
  pipeline_id = pipeline["pipeline_id"]
  print(f"Created pipeline: {pipeline_id}")
  ```


## Step 5: Upload and Process Content

Now, let's upload some content and trigger the processing pipeline:

<CodeGroup>
  ```python Python
  # Upload an object to the bucket
  object = mp.objects.create(
      bucket_id=bucket_id,
      name="sample-content",
      files=[
          {
              "url": "https://example.com/sample-image.jpg",
              "mimetype": "image/jpeg"
          },
          {
              "url": "https://example.com/sample-text.txt",
              "mimetype": "text/plain"
          }
      ]
  )
  
  object_id = object["object_id"]
  print(f"Uploaded object: {object_id}")
  
  # Process the object with our pipeline
  task = mp.pipelines.run(
      pipeline_id=pipeline_id,
      object_id=object_id
  )
  
  task_id = task["task_id"]
  print(f"Processing task started: {task_id}")
  ```


## Step 6: Create a Search Retriever

  ```python Python
      name="content-retriever",
      description="Search across processed content",
      stages=[
          {
              "name": "embedding_search",
              "type": "vector",
              "collection_id": collection_id,
  ```

## Step 7: Search for Content

Now that everything is set up, let's search for content:

<CodeGroup>
  ```python Python
  # Wait for processing to complete
  import time
  time.sleep(30)  # Wait 30 seconds for processing (adjust as needed)
  
  # Search for content by text query
  results = mp.retrievers.search(
      retriever_id=retriever_id,
      query={
          "text": "show me images with people"
      },
      filters={
          "detected_objects": {
              "$contains": "person"
          }
      }
  )
  
  # Print the results
  for result in results["results"]:
      print(f"Found document: {result['document_id']}, Score: {result['score']}")
      print(f"Content: {result['content']}")
  ```


## Complete Example

Here's a complete Python script that ties everything together:

```python
from mixpeek import Mixpeek
import time

# Initialize the Mixpeek client
mp = Mixpeek(api_key="YOUR_API_KEY")

# Create a namespace
namespace = mp.namespaces.create(
    name="quickstart",
    description="My first Mixpeek project"
)
namespace_id = namespace["namespace_id"]

# Create a bucket
bucket = mp.buckets.create(
    namespace_id=namespace_id,
    name="content-bucket",
    description="Storage for our content"
)
bucket_id = bucket["bucket_id"]

# Create a collection
collection = mp.collections.create(
    namespace_id=namespace_id,
    name="content-collection",
    description="Processed content documents"
)
collection_id = collection["collection_id"]

# Create a pipeline
pipeline = mp.pipelines.create(
    namespace_id=namespace_id,
    name="content-pipeline",
    description="Extract features from content",
    input_bucket_id=bucket_id,
    output_collection_ids=[collection_id],
    feature_extractors=[
        {
            "embed": [
                {
                    "type": "url",
                    "embedding_model": "multimodal"
                },
                {
                    "type": "text",
                    "value": "describe this content",
                    "embedding_model": "text"
                }
            ],
            "detect": {
                "faces": {"enabled": True},
                "objects": {"enabled": True}
            },
            "transcribe": {"enabled": True}
        }
    ]
)
pipeline_id = pipeline["pipeline_id"]

# Upload an object
object = mp.objects.create(
    bucket_id=bucket_id,
    name="sample-content",
    files=[
        {
            "url": "https://example.com/sample-image.jpg",
            "mimetype": "image/jpeg"
        }
    ]
)
object_id = object["object_id"]

# Process the object
task = mp.pipelines.run(
    pipeline_id=pipeline_id,
    object_id=object_id
)

# Wait for processing to complete
print("Processing content, please wait...")
time.sleep(30)  # Adjust based on content size

# Create a retriever
retriever = mp.retrievers.create(
    namespace_id=namespace_id,
    name="content-retriever",
    description="Search across processed content",
    stages=[
        {
            "name": "embedding_search",
            "type": "vector",
            "collection_id": collection_id,
            "index": "multimodal",
            "limit": 20
        },
        {
            "name": "filters",
            "type": "filter",
            "input": "embedding_search.results",
            "limit": 10
        }
    ]
)
retriever_id = retriever["retriever_id"]

# Search for content
results = mp.retrievers.search(
    retriever_id=retriever_id,
    query={
        "text": "show me images with people"
    }
)

# Display results
print("\nSearch Results:")
for result in results["results"]:
    print(f"Document ID: {result['document_id']}")
    print(f"Score: {result['score']}")
    print(f"Content: {result['content']}")
    print("---")
```
